# Reduce the levels of rating for quality to 3 levels as high, medium, and low. Assign the levels of 3, 4 to level 0; and 5, 6 to level 1: 7, 8 and 9 to level 2
quality = wine_all_data["quality"].values
rating = []
for num in quality:
    if num<5:
        rating.append("0")
    elif num>6:
        rating.append("2")
    else:
        rating.append("1")
